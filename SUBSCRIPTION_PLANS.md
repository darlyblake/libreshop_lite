## Implémentation Technique

### Vérification des Limites

```typescript
// Exemple de vérification dans storeService
const canAddProduct = (store: Store) => {
  const plan = store.subscription_plan;
  const currentCount = store.product_count || 0;
  
  switch (plan) {
    case 'trial':
      return currentCount < 20;
    case 'standard':
      return currentCount < 50;
    case 'pro':
      return currentCount < 200;
    case 'business':
    case 'enterprise':
      return true; // Illimité
    default:
      return false;
  }
};
```

### Gating des Fonctionnalités

```typescript
// Exemple de vérification d'accès à une fonctionnalité
const hasFeatureAccess = (store: Store, feature: string) => {
  const plan = store.subscription_plan;
  
  const featureMatrix = {
    'analytics_advanced': ['trial', 'pro', 'business', 'enterprise'],
    'loyalty_program': ['trial', 'pro', 'business', 'enterprise'],
    'accounting_advanced': ['trial', 'business', 'enterprise'],
    'multi_store': ['trial', 'business', 'enterprise'],
    'api_access': ['trial', 'business', 'enterprise'],
  };
  
  return featureMatrix[feature]?.includes(plan) || false;
};
```

### Messages d'Upgrade

```typescript
// Message à afficher quand une fonctionnalité n'est pas disponible
const getUpgradeMessage = (currentPlan: string, feature: string) => {
  const messages = {
    'analytics_advanced': 'Cette fonctionnalité nécessite le plan Pro (10 000 FCFA/mois)',
    'loyalty_program': 'Le programme de fidélité est disponible dans le plan Pro',
    'accounting_advanced': 'La comptabilité avancée nécessite le plan Business (25 000 FCFA/mois)',
    'multi_store': 'La gestion multi-boutiques est disponible dans le plan Business',
  };
  
  return messages[feature] || 'Fonctionnalité non disponible avec votre plan actuel';
};
```

## Implémentation Technique

### Service de Gating des Fonctionnalités

Un service complet `featureGatingService.ts` a été créé pour gérer l'accès aux fonctionnalités selon le plan d'abonnement.

#### Structure du Service

```typescript
// Types des fonctionnalités disponibles
export type FeatureKey =
  | 'dashboard_basic'
  | 'dashboard_advanced'
  | 'products_management'
  | 'orders_management'
  | 'pos_caisse'
  | 'analytics_basic'
  | 'analytics_advanced'
  | 'analytics_coach_ai'
  | 'analytics_benchmark'
  | 'clients_basic'
  | 'clients_advanced'
  | 'loyalty_program'
  | 'coupons_basic'
  | 'coupons_unlimited'
  | 'collections_basic'
  | 'collections_unlimited'
  | 'reports_basic'
  | 'reports_advanced'
  | 'reports_detailed'
  | 'export_data'
  | 'low_stock_alerts'
  | 'stock_history'
  | 'returns_management'
  | 'refunds_basic'
  | 'refunds_advanced'
  | 'finance_basic'
  | 'accounting_advanced'
  | 'api_access'
  | 'multi_store'
  | 'support_email'
  | 'support_phone'
  | 'support_24_7'
  | 'white_label'
  | 'custom_development';
```

#### Méthodes Principales

```typescript
class FeatureGatingService {
  // Vérifier si une boutique a accès à une fonctionnalité
  hasFeatureAccess(store: Store, featureKey: FeatureKey): boolean

  // Vérifier si une boutique peut ajouter un produit
  canAddProduct(store: Store, currentProductCount: number): boolean

  // Vérifier si une boutique peut créer une nouvelle boutique
  canAddStore(store: Store, currentStoreCount: number): boolean

  // Obtenir les limites d'un plan
  getPlanLimits(plan: string): PlanLimits

  // Obtenir toutes les fonctionnalités d'un plan
  getPlanFeatures(plan: string): FeatureKey[]

  // Obtenir le message d'upgrade pour une fonctionnalité
  getUpgradeMessage(featureKey: FeatureKey): string

  // Obtenir le plan minimum requis pour une fonctionnalité
  getRequiredPlanForFeature(featureKey: FeatureKey): string

  // Activer une fonctionnalité pour un plan (admin)
  enableFeatureForPlan(featureKey: FeatureKey, plan: string): void

  // Désactiver une fonctionnalité pour un plan (admin)
  disableFeatureForPlan(featureKey: FeatureKey, plan: string): void
}
```

#### Utilisation dans les Écrans

```typescript
import { featureGatingService } from '../services/featureGatingService';

// Vérifier l'accès à une fonctionnalité
if (featureGatingService.hasFeatureAccess(store, 'analytics_advanced')) {
  // Afficher les analytics avancés
} else {
  // Afficher un message d'upgrade
  Alert.alert(
    'Fonctionnalité Premium',
    featureGatingService.getUpgradeMessage('analytics_advanced')
  );
}

// Vérifier si on peut ajouter un produit
if (!featureGatingService.canAddProduct(store, currentProductCount)) {
  Alert.alert(
    'Limite atteinte',
    'Vous avez atteint la limite de produits pour votre plan. Passez à un plan supérieur pour en ajouter plus.'
  );
}
```

### Stratégie de Migration entre Plans

#### Détection des Limites

```typescript
// Dans les écrans de création/édition
const checkProductLimit = () => {
  const store = await storeService.getByUser(userId);
  const currentCount = await productService.countByStore(store.id);

  if (!featureGatingService.canAddProduct(store, currentCount)) {
    const limits = featureGatingService.getPlanLimits(store.subscription_plan);
    Alert.alert(
      'Limite atteinte',
      `Votre plan ${store.subscription_plan} permet ${limits.maxProducts} produits maximum. ` +
      `Passez au plan ${getNextPlan(store.subscription_plan)} pour continuer.`
    );
    return false;
  }
  return true;
};
```

#### Messages d'Upgrade Automatiques

```typescript
const getNextPlan = (currentPlan: string): string => {
  const hierarchy = ['trial', 'standard', 'pro', 'business', 'enterprise'];
  const currentIndex = hierarchy.indexOf(currentPlan);
  return hierarchy[currentIndex + 1] || 'enterprise';
};

const showUpgradePrompt = (featureKey: FeatureKey) => {
  const requiredPlan = featureGatingService.getRequiredPlanForFeature(featureKey);
  const price = featureGatingService.getPlanPrice(requiredPlan);

  Alert.alert(
    'Upgrade requis',
    `${featureGatingService.getUpgradeMessage(featureKey)}\n\n` +
    `Prix: ${price > 0 ? `${price.toLocaleString()} FCFA/mois` : 'Sur devis'}`,
    [
      { text: 'Annuler', style: 'cancel' },
      {
        text: 'Voir les plans',
        onPress: () => navigation.navigate('SellerSubscriptions')
      }
    ]
  );
};
```

#### Migration Automatique

```typescript
// Dans le service de paiement
const handleSubscriptionUpgrade = async (storeId: string, newPlan: string) => {
  const store = await storeService.getById(storeId);
  const oldPlan = store.subscription_plan;

  // Vérifier si la migration est nécessaire
  if (oldPlan !== newPlan) {
    // Sauvegarder l'ancien plan pour rollback si nécessaire
    await storeService.update(storeId, {
      subscription_plan: newPlan,
      previous_plan: oldPlan,
      migration_date: new Date().toISOString(),
    });

    // Notifier l'utilisateur
    await notificationService.send(store.user_id, {
      title: 'Abonnement mis à jour',
      body: `Votre boutique est passée du plan ${oldPlan} au plan ${newPlan}`,
    });
  }
};
```

#### Rollback en Cas d'Erreur

```typescript
const rollbackSubscription = async (storeId: string) => {
  const store = await storeService.getById(storeId);

  if (store.previous_plan) {
    await storeService.update(storeId, {
      subscription_plan: store.previous_plan,
      previous_plan: null,
    });

    await notificationService.send(store.user_id, {
      title: 'Erreur de paiement',
      body: 'Votre abonnement a été rétabli au plan précédent',
    });
  }
};
```

## Conclusion

Cette structure de plans d'abonnement permet de :

1. **Acquisition:** Le plan gratuit attire de nouveaux vendeurs
2. **Conversion:** Les limites progressives encouragent les upgrades
3. **Rétention:** Les fonctionnalités exclusives maintiennent les clients premium
4. **Revenus:** Pricing échelonné pour différents segments de marché

Les fonctionnalités sont stratégiquement distribuées pour créer une valeur claire à chaque niveau d'abonnement.
